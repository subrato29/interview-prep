Bubble sort
----------------------------------------------------------------------------
----------------
algo
----------------
begin BubbleSort(list)

   for all elements of list
      if list[i] > list[i+1]
         swap(list[i], list[i+1])
      end if
   end for
   
   return list
   
end BubbleSort
------------------
Pseudo code
------------------
int[] arr = new int[] { 6, 8, 7, 4, 312, 78, 54, 9, 12, 100, 89, 74 };

for (int i = 0; i < arr.length; i++) {
        for (int j = i + 1; j < arr.length; j++) {
            int tmp = 0;
            if (arr[i] > arr[j]) {
	   	tmp = arr[i];
 		arr[i] = arr[j];
       		arr[j] = tmp;
	            }
	}
      }
	for(int i=0; i < arr.length; i++)	{  
		syso(arr[i] + " ");  
	}	
}
